Пример разработан в среде Borland Developer Studio 2006. Создайте новый проект: "File - New - Other". В появившемся диалоге выберите "C++Builder Projects - ActiveX - ActiveX Library". Перед началом реализации импортируйте в проект библиотеку типов GrymCore. Для этого выберите пункт меню "Component - Import Component". В появившемся диалоге выберите строку "Import a Type Library" и нажмите "Next". Дальше найдите в списке "GrymCore 1.0 Type Library". Если библиотека отсутствует, запустите приложение ДубльГИС v3 и попробуйте еще раз. В следующем диалоге убедитесь, что "Palette Page" стоит "(none)", выберите путь, куда разместить создаваемый файл "Unit Dir Name", нажмите "Next". В новом диалоге выберите строку "Add unit to (ваше имя проекта) project" и нажмите "Finish". После сохранения проекта добавьте новый файл: "File - New - Other". В появившемся диалоге выберите "C++Builder - ActiveX - COM Object".

Файл проекта

Объявляется внешняя функция CreateGrymPlugin. 

#include <vcl.h>
#pragma hdrstop
#include <atl\atlvcl.h>
#include "MyPlugin.h"
#pragma package(smart_init)

TComModule  MyModule;
TComModule &_Module = MyModule;

BEGIN_OBJECT_MAP(ObjectMap)
  OBJECT_ENTRY(CLSID_MyPlugin, TMyPlugin)
END_OBJECT_MAP()

int WINAPI DllEntryPoint(HINSTANCE hinst, unsigned long reason, void*)
{
    if (reason == DLL_PROCESS_ATTACH)
    {
        _Module.Init(ObjectMap, hinst);
	DisableThreadLibraryCalls(hinst);
    }
    return TRUE;
}

void ModuleTerm(void) {
    _Module.Term();
}
#pragma exit ModuleTerm 63

//------------------------------------------------------------------------------

STDAPI __export DllRegisterServer(void) {
    return _Module.RegisterServer(TRUE);
}

STDAPI __export DllUnregisterServer(void) {
    return _Module.UnregisterServer();
}

STDAPI __export DllCanUnloadNow(void) {
    return (_Module.GetLockCount()==0) ? S_OK : S_FALSE;
}

STDAPI __export DllGetClassObject(REFCLSID rclsid, REFIID riid, LPVOID* ppv) {
    return _Module.GetClassObject(rclsid, riid, ppv);
}

//---------------------------------------------------------------------------

STDAPI __export CreateGrymPlugin(IUnknown **pPlugin)
{
    CComObject<TMyPlugin> *pMyPlugin;
    CComObject<TMyPlugin>::CreateInstance(&pMyPlugin);
    if (pMyPlugin == NULL) return S_FALSE;
    pMyPlugin->AddRef();
    *pPlugin = pMyPlugin->GetUnknown();
    return S_OK;
}

Используются интерфейсы IGrymPlugin, IGrymPluginInfo, IGrymPluginOptions. 

#ifndef MyPluginH
#define MyPluginH

#define ATL_APARTMENT_THREADED

#include "MyModule_TLB.h"
#include "GrymCore_TLB.h"

class ATL_NO_VTABLE TMyPlugin :
    public CComObjectRootEx<CComSingleThreadModel>,
    public CComCoClass<TMyPlugin, &CLSID_MyPlugin>,
    public Grymcore_tlb::IGrymPlugin,
    public Grymcore_tlb::IGrymPluginInfo,
    public Grymcore_tlb::IGrymPluginOptions
{
public:
    TMyPlugin(){};

    DECLARE_THREADING_MODEL(otApartment);
    DECLARE_PROGID("MyModule.MyPlugin");
    DECLARE_DESCRIPTION("");

    static HRESULT WINAPI UpdateRegistry(BOOL bRegister)
    {
	TTypedComServerRegistrarT<TMyPlugin>
	regObj(GetObjectCLSID(), GetProgID(), GetDescription());
	return regObj.UpdateRegistry(bRegister);
    }

BEGIN_COM_MAP(TMyPlugin)
    COM_INTERFACE_ENTRY(IGrymPlugin)
    COM_INTERFACE_ENTRY(IGrymPluginInfo)
    COM_INTERFACE_ENTRY(IGrymPluginOptions)
END_COM_MAP()

public:
    // IGrymPlugin methods
    HRESULT __stdcall Initialize(IGrym * pRoot, IBaseViewThread * pBaseView);
    HRESULT __stdcall Terminate();
    // IGrymPluginInfo methods
    HRESULT __stdcall get_Name(BSTR *pName);
    HRESULT __stdcall get_Description(BSTR *pDesc);
    HRESULT __stdcall get_Copyright(BSTR * pVal);
    HRESULT __stdcall get_Icon(OLE_HANDLE *pVal);
    // IGrymPluginOptions methods
    HRESULT __stdcall OptionDialog(VARIANT_BOOL *pRet);
};

#endif // MyPluginH
