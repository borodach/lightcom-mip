Пример разработан в среде Borland Developer Studio 2006. Создайте новый проект: File-New-Other - Delphi_Projects-ActiveX-ActiveX_Library. Импортируйте библиотеку типов GrymCore в проект. Добавьте два COM-объекта (MyPluginMain, MyCommand): File-New-Other - Delphi_Projects-ActiveX-COM_Object.


Перейдите на окно редактирования библиотеки типов MyPlugin.tlb. Добавьте в секцию Uses библиотеку GrymCore 1.0 Type Library. Удалите созданные интерфейсы из библиотеки и из секции Implements классов. Добавьте в секцию Implements класса MyPluginMain интерфейсы IGrymPlugin, IGrymPluginInfo, IGrymPluginOptions. В качестве интерфейса по-умолчанию постаьте IGrymPlugin. Добавьте в секцию Implements класса MyCommand интерфейсы ICommandAction, ICommandAccelerator, ICommandAppearance, ICommandPlacement, ICommandState. В качестве интерфейса по-умолчанию поставьте ICommandAction.


В примере показана возможность подключения модуля через COM-категорию. После регистрации ActiveX-сервера модуль будет подключен автоматически при загрузке карты. В этом случае файл *.DLL не нужно копировать в папку Plugins приложения.

Главный файл библиотеки

library MyPlugin;

uses
  ComServ,
  MyPlugin_TLB in 'MyPlugin_TLB.pas',
  GrymCore_TLB in 'GrymCore_TLB.pas',
  UMyPluginMain in 'UMyPluginMain.pas' {MyPluginMain: CoClass},
  UMyCommand in 'UMyCommand.pas' {MyCommand: CoClass};

exports
  DllGetClassObject,
  DllCanUnloadNow,
  // Эта функция вызывается при регистрации ActiveX сервера.
  // Перекрываем функцию и регистрируемся в
  // COM-категории "2gis Grym Plugins".
  UMyPluginMain.DllRegisterServer,
  // Перекрываем функцию для удаления из категории.
  UMyPluginMain.DllUnregisterServer,
  // Эта функция вызывается, если файл "MyPlugin.dll"
  // находится в папке "Plugins" приложения.
  UMyPluginMain.CreateGrymPlugin;

{$R *.TLB}

{$R *.RES}
{$R MyRes.res}  // Ресурс иконки плагина c именем "PLUGICON".

begin
end.

Файл объекта модуля

unit UMyPluginMain;

{$WARN SYMBOL_PLATFORM OFF}

interface

uses
  Windows, ActiveX, SysUtils, Classes, ComObj, StdVcl,
  MyPlugin_TLB, GrymCore_TLB;

type
  TMyPluginMain = class(TTypedComObject, IGrymPlugin, IGrymPluginInfo,
    IGrymPluginOptions)
  private
    function GP_Initialize(const pRoot: IGrym;
      const pBaseView: IBaseViewthread): HRESULT; stdcall;
  protected
    // IGrymPlugin methods.
    // Функция обозначена таким образом, потому что декларация
    // Initialize() перекрывает одноименный метод TComObject.
    function IGrymPlugin.Initialize = GP_Initialize;
    function Terminate: HRESULT; stdcall;

    // IGrymPluginInfo methods.
    function Get_Name(out pName: WideString): HRESULT; stdcall;
    function Get_Description(out pDesc: WideString): HRESULT; stdcall;
    function Get_CopyRight(out pVal: WideString): HRESULT; stdcall;
    function Get_Icon(out pVal: OLE_HANDLE): HRESULT; stdcall;

    // IGrymPluginOptions methods.
    function OptionDialog(out pRet: WordBool): HRESULT; stdcall;
  end;

function DllRegisterServer: HRESULT; stdcall;
function DllUnregisterServer: HRESULT; stdcall;
function CreateGrymPlugin(var pPlugin: IUnknown): HRESULT; stdcall;

implementation

uses
  UMyCommand, ComServ;

const
  // GUID категории "2gis Grym Plugins".
  DGIS_CAT: TGUID = '{687FCBEC-C7A2-4D4C-B79B-DA650E7C29F7}';

function DllRegisterServer: HRESULT; stdcall;
var
  pCatRegister: ICatRegister;
begin
  // Вызываем готовую функцию регистрации ActiveX-сервера.
  result := ComServ.DllRegisterServer;

  if result = S_OK then begin
    // Регистрируем класс в COM-категории "2gis Grym Plugins".
    CoCreateInstance(CLSID_StdComponentCategoryMgr, nil, CLSCTX_ALL,
      ICatRegister, pCatRegister);
    if pCatRegister <> nil then begin
      pCatRegister.RegisterClassImplCategories(CLASS_MyPluginMain, 1, @DGIS_CAT);
    end;
  end;
end;

function DllUnregisterServer: HRESULT; stdcall;
var
  pCatRegister: ICatRegister;
begin
  // Удаляем регистрацию в категории.
  CoCreateInstance(CLSID_StdComponentCategoryMgr, nil, CLSCTX_ALL,
    ICatRegister, pCatRegister);
  if pCatRegister <> nil then begin
    pCatRegister.UnregisterClassImplCategories(CLASS_MyPluginMain, 1, @DGIS_CAT);
  end;
  // Вызываем готовую функцию удаления регистрации ActiveX-сервера.
  result := ComServ.DllUnregisterServer;
end;

function CreateGrymPlugin(var pPlugin: IUnknown): HRESULT; stdcall;
begin
  result := S_FALSE;
  try
    pPlugin := TMyPluginMain.Create;
    if pPlugin <> nil then result := S_OK;
  except
  end;
end;

{ TMyPluginMain }

function TMyPluginMain.GP_Initialize(const pRoot: IGrym;
  const pBaseView: IBaseViewThread): HRESULT;
begin
  result := UMyCommand.AddCommandIntoMenu(pBaseView);
end;

function TMyPluginMain.Terminate: HRESULT;
begin
  result := S_OK;
end;

function TMyPluginMain.Get_Name(out pName: WideString): HRESULT;
begin
  pName := 'Test plugin';
  result := S_OK;
end;

function TMyPluginMain.Get_Description(out pDesc: WideString): HRESULT;
begin
  pDesc := 'Test plugin description';
  result := S_OK;
end;

function TMyPluginMain.Get_CopyRight(out pVal: WideString): HRESULT;
begin
  pVal := 'Freeware';
  result := S_OK;
end;

function TMyPluginMain.Get_Icon(out pVal: OLE_HANDLE): HRESULT;
begin
  pVal := LoadIcon(SysInit.HInstance, 'PLUGICON');
  if pVal <> 0 then result := S_OK else result := S_FALSE;
end;

function TMyPluginMain.OptionDialog(out pRet: WordBool): HRESULT;
begin
  MessageBox(0, 'OptionDialog', 'Test plugin', MB_OK);
  pRet := TRUE;
  result := S_OK;
end;

initialization
  TTypedComObjectFactory.Create(ComServer, TMyPluginMain, Class_MyPluginMain,
    ciMultiInstance, tmApartment);

end.

Файл объекта команды

unit UMyCommand;

{$WARN SYMBOL_PLATFORM OFF}

interface

uses
  Windows, ActiveX, SysUtils, Classes, ComObj, StdVcl,
  MyPlugin_TLB, GrymCore_TLB;

type
  TMyCommand = class(TTypedComObject, ICommandAction, ICommandAccelerator,
    ICommandAppearance, ICommandPlacement, ICommandState)
  protected
    // ICommandAction methods.
    function OnCommand(const pContext: IContextBase): HRESULT; stdcall;

    // ICommandAccelerator methods.
    function Get_Accelerator(out pVal: Integer): HRESULT; stdcall;

    // ICommandAppearance methods.
    function Get_ImageName(out pVal: WideString): HRESULT; stdcall;
    function Get_DefaultBitmap(const bsType: WideString;
      out pVal: OLE_HANDLE): HResult; stdcall;
    function Get_Title(out pVal: WideString): HRESULT; stdcall;

    // ICommandPlacement methods.
    function Get_GroupName(out pVal: WideString): HRESULT; stdcall;
    function Get_Priority(out pVal: Integer): HRESULT; stdcall;
    function Get_Position(out pVal: Integer): HRESULT; stdcall;

    // ICommandState methods.
    function Get_Available(const pContext: IContextBase;
      out pVal: WordBool): HRESULT; stdcall;
    function Get_Enabled(const pContext: IContextBase;
      out pVal: WordBool): HRESULT; stdcall;
    function Get_Checked(const pContext: IContextBase;
      out pVal: WordBool): HRESULT; stdcall;
  end;

function AddCommandIntoMenu(const fView: IBaseViewThread): HRESULT;

implementation

uses
  UMyPluginMain, ComServ;

function AddCommandIntoMenu(const fView: IBaseViewThread): HRESULT;
var
  pCommand: ICommandAction;
  pMap: IMap;
  pPopupMenu: IPopupMenu;
begin
  result := fView.Get_Map(pMap);
  if result = S_OK then begin
    result := pMap.Get_ContextMenu(pPopupMenu);
    if result = S_OK then begin
      result := E_FAIL;
      try
        pCommand := TMyCommand.Create;
        if pCommand <> nil then
          result := pPopupMenu.AddCommand(pCommand);
      except
      end;
    end;
  end;
end;

function TMyCommand.OnCommand(const pContext: IContextBase): HRESULT;
var
  pMapPos: IMapPoint;
  X, Y: Double;
begin
  result := IPopupMenuMapViewContext(pContext).Get_MapPos(pMapPos);
  if result = S_OK then begin
    result := pMapPos.Get_X(X);
    if result = S_OK then begin
      result := pMapPos.Get_Y(Y);
      if result = S_OK then begin
        MessageBox(0,
          PCHAR('X = ' + FloatToStr(X) + ', Y = ' + FloatToStr(Y)),
          'Test plugin', MB_OK);
      end;
    end;
  end;
end;

function TMyCommand.Get_Accelerator(out pVal: Integer): HRESULT;
begin
  result := S_FALSE;
end;

function TMyCommand.Get_ImageName(out pVal: WideString): HRESULT;
begin
  result := S_FALSE;
end;

function TMyCommand.Get_DefaultBitmap(const bsType: WideString;
  out pVal: OLE_HANDLE): HRESULT;
begin
  result := S_FALSE;
end;

function TMyCommand.Get_Title(out pVal: WideString): HRESULT;
begin
  pVal := 'Show Coords';
  result := S_OK;
end;

function TMyCommand.Get_GroupName(out pVal: WideString): HRESULT;
begin
  pVal := 'GR_0001';
  result := S_OK;
end;

function TMyCommand.Get_Priority(out pVal: Integer): HRESULT;
begin
  result := S_FALSE;
end;

function TMyCommand.Get_Position(out pVal: Integer): HRESULT;
begin
  pVal := 1;
  result := S_OK;
end;

function TMyCommand.Get_Available(const pContext: IContextBase;
  out pVal: WordBool): HRESULT;
begin
  pVal := TRUE;
  result := S_OK;
end;

function TMyCommand.Get_Enabled(const pContext: IContextBase;
  out pVal: WordBool): HRESULT;
begin
  pVal := TRUE;
  result := S_OK;
end;

function TMyCommand.Get_Checked(const pContext: IContextBase;
  out pVal: WordBool): HRESULT;
begin
  pVal := FALSE;
  result := S_OK;
end;

initialization
  TTypedComObjectFactory.Create(ComServer, TMyCommand, Class_MyCommand,
    ciMultiInstance, tmApartment);
end.

Файл библиотеки типов

Библиотека генерируется средой автоматически. В примере - без комментариев.

unit MyPlugin_TLB;

interface

uses Windows, ActiveX, Classes, Graphics, GrymCore_TLB, StdVCL, Variants;

const
  MyPluginMajorVersion = 1;
  MyPluginMinorVersion = 0;

  LIBID_MyPlugin: TGUID = '{E63693BD-B49B-4589-8381-5BE555DF3A21}';

  CLASS_MyPluginMain: TGUID = '{A7A5E5EC-DE63-47F3-92E6-7B80FF6B7D73}';
  CLASS_MyCommand: TGUID = '{469EAA25-70A8-4270-BC03-3889AE435ED2}';

type
  MyPluginMain = IGrymPlugin;
  MyCommand = ICommandAction;

  CoMyPluginMain = class
    class function Create: IGrymPlugin;
    class function CreateRemote(const MachineName: string): IGrymPlugin;
  end;

  CoMyCommand = class
    class function Create: ICommandAction;
    class function CreateRemote(const MachineName: string): ICommandAction;
  end;

implementation

uses ComObj;

class function CoMyPluginMain.Create: IGrymPlugin;
begin
  Result := CreateComObject(CLASS_MyPluginMain) as IGrymPlugin;
end;

class function CoMyPluginMain.CreateRemote(const MachineName: string): IGrymPlugin;
begin
  Result := CreateRemoteComObject(MachineName, CLASS_MyPluginMain) as IGrymPlugin;
end;

class function CoMyCommand.Create: ICommandAction;
begin
  Result := CreateComObject(CLASS_MyCommand) as ICommandAction;
end;

class function CoMyCommand.CreateRemote(const MachineName: string): ICommandAction;
begin
  Result := CreateRemoteComObject(MachineName, CLASS_MyCommand) as ICommandAction;
end;

end.
