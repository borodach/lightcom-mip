Пример разработан в среде MSVC 2003.

Создаем новый проект: File — New — Project..., выбираем ATL Project. Снимаем установку Attributed.

В настройках проекта, в разделе C/C++ — Code Generation изменяем Runtime Library: выбираем Multi-threaded Debug и Multi-threaded для конфигураций Debug и Release соответственно.

В файл stdafx.h добавляем строчку для импорта библиотеки типов: 

// Grym.exe
#import "libid:7AA02C95-0B4A-43aa-92D8-BA40511A7F3F"

Создаем новый класс, который будет использоваться как главный класс создаваемого модуля: Project — Add class..., выбираем Generic C++ Class. Указываем название класса "PluginImpl", остальные поля будут заполнены автоматически.

Добавляем в главный файл проекта функцию CreateGrymPlugin.

В файл *.def добавляем строчку для экспорта этой функции: 

  CreateGrymPlugin  PRIVATE

// главная точка входа в плагин в случае если не используем ригистрацию в категории
extern "C" int __stdcall CreateGrymPlugin( IUnknown **pPlugin )
{
  GrymCore::IGrymPluginPtr plugin = PluginImpl::CreateInstance();
  if ( plugin == 0 )
  {
    return 1;
  }
  *pPlugin = IUnknownPtr(plugin).Detach();
  return 0;
}
Заголовочный файл COM-объекта
#pragma once

class ATL_NO_VTABLE PluginImpl : 
    public ATL::CComObjectRootEx<ATL::CComSingleThreadModel>,
    public GrymCore::IGrymPlugin,
    public GrymCore::IGrymPluginInfo
{
public:
    static GrymCore::IGrymPluginPtr CreateInstance();
protected:
    PluginImpl() {}

BEGIN_COM_MAP(PluginImpl)
    COM_INTERFACE_ENTRY(GrymCore::IGrymPlugin)
    COM_INTERFACE_ENTRY(GrymCore::IGrymPluginInfo)
END_COM_MAP()

public: // GrymCore::IGrymPlugin
    STDMETHOD(raw_Initialize)(GrymCore::IGrym *pRoot, GrymCore::IBaseViewThread *pBaseView);
    STDMETHOD(raw_Terminate)();
public: // GrymCore::IGrymPluginInfo
    STDMETHOD(get_Name)(BSTR *pName);
    STDMETHOD(get_Description)(BSTR *pDesc);
    STDMETHOD(get_Copyright)(BSTR *pVal);
    STDMETHOD(get_Icon)(OLE_HANDLE *pVal);
private:
    GrymCore::IGrymPtr grym_root_;
    GrymCore::IBaseViewThreadPtr base_view_;
};
Файл реализации
#include "StdAfx.h"
#include ".\pluginimpl.h"

GrymCore::IGrymPluginPtr PluginImpl::CreateInstance()
{
    ATL::CComObject<PluginImpl> *obj;
    ATLVERIFY(S_OK == ATL::CComObject<PluginImpl>::CreateInstance(&obj));
    return obj;
}

STDMETHODIMP PluginImpl::raw_Initialize(GrymCore::IGrym *pRoot, GrymCore::IBaseViewThread *pBaseView)
{
    try
    {
	grym_root_ = pRoot;
	base_view_ = pBaseView;
	return S_OK;
    } catch(...){}
	return E_FAIL;
}

STDMETHODIMP PluginImpl::raw_Terminate()
{
    try
    { // необходимо освободить все занятые ссылки, чтобы поток работы с базой корректно закрылся
	grym_root_ = NULL;
	base_view_ = NULL;
    }catch(...){}
	return S_OK;
}

STDMETHODIMP PluginImpl::get_Name(BSTR *pName)
{
    try
    {
	if (!pName) return E_POINTER;
	*pName = 0;
	static const _bstr_t name = OLESTR("SamplePlugin");
	*pName = name.copy();
	return S_OK;
    }catch(...){}
	return E_FAIL;
}

STDMETHODIMP PluginImpl::get_Description(BSTR *pDesc)
{
    try
    {
	if (!pDesc) return E_POINTER;
	*pDesc = 0;
	static const _bstr_t desc = OLESTR("SamplePlugin description. Runing on: ");
	*pDesc = (desc + base_view_->GetBaseReference()->GetFullName()).copy();
	return S_OK;
    }catch(...){}
	return E_FAIL;
}

STDMETHODIMP PluginImpl::get_Copyright(BSTR *pVal)
{
    try
    {
	if (!pVal) return E_POINTER;
	*pVal = 0;
	static const _bstr_t txt = OLESTR("2007, 2GIS Plugin Team");
	*pVal = txt.copy();
	return S_OK;
    }catch(...){}
	return E_FAIL;
}

STDMETHODIMP PluginImpl::get_Icon(OLE_HANDLE *pVal)
{
    try
    {
	if (!pVal) return E_POINTER;
	// We have no icon
	*pVal = 0;
	return S_FALSE;
    }catch(...){}
	return E_FAIL;
}
